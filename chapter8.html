1. Introduction to Functions

A function is a block of code that performs a specific task. Functions promote code reusability and modular programming.

Example 1: Simple Function


#include <stdio.h>

void greet() {
    printf("Hello, welcome to C programming!\n");
}

int main() {
    greet();
    return 0;
}

Example 2: Function with Return Value


#include <stdio.h>

int add(int a, int b) {
    return a + b;
}

int main() {
    int result = add(5, 7);
    printf("Result: %d\n", result);
    return 0;
}

2. Types of Functions

  • Built-in Functions: Predefined functions in C like printf(), scanf(), strcpy(), etc.
  • User-defined Functions: Functions created by the user.

Example 1: Using Built-in Function


#include <stdio.h>

int main() {
    printf("This is a built-in function\n");
    return 0;
}

Example 2: User-Defined Function


#include <stdio.h>

void displayMessage() {
    printf("This is a user-defined function\n");
}

int main() {
    displayMessage();
    return 0;
}

Example 3: Passing Values to User-Defined Function


#include <stdio.h>

void square(int num) {
    printf("Square of %d is %d\n", num, num * num);
}

int main() {
    square(4);
    square(9);
    return 0;
}

Example 4: User-Defined Function with Return Value


#include <stdio.h>

int multiply(int a, int b) {
    return a * b;
}

int main() {
    int result = multiply(4, 5);
    printf("Result: %d\n", result);
    return 0;
}

3. Function Prototypes

A function prototype is a declaration of a function that informs the compiler about its return type and parameters before its actual definition.

Example 1: Function Prototype


#include <stdio.h>

// Function prototype
int add(int a, int b);

int main() {
    int sum = add(4, 7);
    printf("Sum: %d\n", sum);
    return 0;
}

// Function definition
int add(int a, int b) {
    return a + b;
}

Example 2: Prototype with Different Return Type


#include <stdio.h>

float divide(int a, int b);  // Prototype

int main() {
    float result = divide(10, 3);
    printf("Division result: %.2f\n", result);
    return 0;
}

float divide(int a, int b) {
    return (float)a / b;
}

4. Parameter Passing

There are two ways to pass parameters to a function in C: pass by value and pass by reference.

Pass by Value

In pass by value, the actual value is passed to the function, and changes made inside the function do not affect the original variable.

Example: Pass by Value (continued)


#include <stdio.h>

void increment(int num) {
    num++;
    printf("Inside function: %d\n", num);
}

int main() {
    int value = 10;
    increment(value);
    printf("Outside function: %d\n", value);
    return 0;
}

Pass by Reference

In pass by reference, the address of the variable is passed, allowing the function to modify the original variable.

Example: Pass by Reference


#include <stdio.h>

void increment(int *num) {
    (*num)++;
}

int main() {
    int value = 10;
    increment(&value);
    printf("After increment: %d\n", value);
    return 0;
}

5. Recursion

Recursion occurs when a function calls itself. It is useful for solving problems that can be broken down into smaller, similar problems.

Example 1: Factorial Calculation


#include <stdio.h>

int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

int main() {
    int number = 5;
    printf("Factorial of %d is %d\n", number, factorial(number));
    return 0;
}

Example 2: Fibonacci Sequence


#include <stdio.h>

int fibonacci(int n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() {
    int terms = 5;
    for (int i = 0; i < terms; i++) {
        printf("%d ", fibonacci(i));
    }
    printf("\n");
    return 0;
}

Example 3: Power of a Number


#include <stdio.h>

int power(int base, int exp) {
    if (exp == 0) return 1;
    return base * power(base, exp - 1);
}

int main() {
    int base = 2, exponent = 3;
    printf("%d^%d = %d\n", base, exponent, power(base, exponent));
    return 0;
}

Example 4: Sum of Natural Numbers


#include <stdio.h>

int sum(int n) {
    if (n <= 1) return n;
    return n + sum(n - 1);
}

int main() {
    int number = 5;
    printf("Sum of first %d natural numbers is %d\n", number, sum(number));
    return 0;
}
