1. Introduction to Pointers

A pointer in C is a variable that stores the memory address of another variable. Pointers are used to dynamically allocate memory, handle arrays and functions, and manipulate data efficiently.

To declare a pointer, use the `*` symbol:


#include <stdio.h>

int main() {
    int num = 10;
    int *ptr = # // Pointer to integer

    printf("Value of num: %d\n", num);
    printf("Address of num: %p\n", (void*)&num);
    printf("Value of ptr: %p\n", (void*)ptr);
    printf("Value pointed to by ptr: %d\n", *ptr);

    return 0;
}

2. Pointer Operations

Dereferencing Pointers

Dereferencing a pointer means accessing the value stored at the address the pointer is pointing to:


#include <stdio.h>

int main() {
    int num = 20;
    int *ptr = #

    printf("Value of num: %d\n", *ptr); // Dereferencing ptr to get value of num
    return 0;
}

Pointer Arithmetic

Pointers can be incremented or decremented to navigate through arrays:


#include <stdio.h>

int main() {
    int arr[5] = {1, 2, 3, 4, 5};
    int *ptr = arr;

    for (int i = 0; i < 5; i++) {
        printf("Element %d: %d\n", i, *(ptr + i)); // Pointer arithmetic to access array elements
    }

    return 0;
}

3. Pointers and Functions

Passing Pointers to Functions

Pointers can be used to pass large data structures to functions efficiently:


#include <stdio.h>

void increment(int *num) {
    (*num)++;
}

int main() {
    int value = 10;
    increment(&value); // Passing address of value
    printf("Incremented value: %d\n", value);
    return 0;
}

Function Pointers

Function pointers are used to store the address of a function and call it indirectly:


#include <stdio.h>

void sayHello() {
    printf("Hello!\n");
}

int main() {
    void (*funcPtr)() = sayHello; // Function pointer
    funcPtr(); // Calling function via pointer
    return 0;
}

4. Examples

Example 1: Pointer Basics


#include <stdio.h>

int main() {
    int value = 50;
    int *ptr = &value;

    printf("Value: %d\n", *ptr); // Access value through pointer
    return 0;
}

Example 2: Pointer with Arrays


#include <stdio.h>

int main() {
    int numbers[3] = {1, 2, 3};
    int *ptr = numbers;

    for (int i = 0; i < 3; i++) {
        printf("Element %d: %d\n", i, *(ptr + i));
    }

    return 0;
}

Example 3: Dynamic Memory Allocation


#include <stdio.h>
#include <stdlib.h>

int main() {
    int *ptr = (int *)malloc(sizeof(int) * 5); // Allocate memory for 5 integers

    if (ptr == NULL) {
        printf("Memory allocation failed\n");
        return 1;
    }

    for (int i = 0; i < 5; i++) {
        ptr[i] = i * 10;
    }

    for (int i = 0; i < 5; i++) {
        printf("Element %d: %d\n", i, ptr[i]);
    }

    free(ptr); // Free allocated memory
    return 0;
}

Example 4: Swapping Two Numbers Using Pointers


#include <stdio.h>

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x = 5, y = 10;
    swap(&x, &y);
    printf("x: %d, y: %d\n", x, y);
    return 0;
}

Example 5: Pointer to Pointer


#include <stdio.h>

int main() {
    int value = 100;
    int *ptr = &value;
    int **ptr2 = &ptr;

    printf("Value: %d\n", **ptr2); // Access value using pointer to pointer
    return 0;
}
