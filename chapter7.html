1. Introduction to Structures

A structure in C is a user-defined data type that groups related variables of different types under a single name. Structures are used to represent a record or a collection of different types of data.

To define a structure, use the `struct` keyword:


#include <stdio.h>

struct Person {
    char name[50];
    int age;
    float height;
};

int main() {
    struct Person person1 = {"Alice", 30, 5.6};

    printf("Name: %s\n", person1.name);
    printf("Age: %d\n", person1.age);
    printf("Height: %.2f\n", person1.height);

    return 0;
}

2. Nested Structures

Structures can contain other structures, allowing for complex data representation:


#include <stdio.h>

struct Address {
    char street[100];
    char city[50];
};

struct Person {
    char name[50];
    struct Address address;
};

int main() {
    struct Person person1 = {"Bob", {"123 Main St", "Springfield"}};

    printf("Name: %s\n", person1.name);
    printf("Address: %s, %s\n", person1.address.street, person1.address.city);

    return 0;
}

3. Arrays of Structures

You can create an array of structures to handle multiple records efficiently:


#include <stdio.h>

struct Student {
    char name[50];
    int rollNo;
};

int main() {
    struct Student students[3] = {
        {"John", 101},
        {"Jane", 102},
        {"Doe", 103}
    };

    for (int i = 0; i < 3; i++) {
        printf("Student %d: %s, Roll No: %d\n", i + 1, students[i].name, students[i].rollNo);
    }

    return 0;
}

4. Introduction to Unions

A union is a user-defined data type similar to a structure, but it allows storing different data types in the same memory location. Only one member can hold a value at any given time.

To define a union, use the `union` keyword:


#include <stdio.h>

union Data {
    int intValue;
    float floatValue;
    char charValue;
};

int main() {
    union Data data;

    data.intValue = 10;
    printf("Integer: %d\n", data.intValue);

    data.floatValue = 220.5;
    printf("Float: %.2f\n", data.floatValue);

    data.charValue = 'A';
    printf("Character: %c\n", data.charValue);

    return 0;
}

5. Structures vs Unions

Here are the main differences between structures and unions:

  • Memory Allocation: Structures allocate memory for all members simultaneously, while unions allocate memory for the largest member only.
  • Access: Structures can access all members at any time, while unions can only access one member at a time.
  • Use Cases: Structures are used for grouping related data, whereas unions are used to store different types of data in the same location to save memory.

6. Examples

Example 1: Basic Structure


#include <stdio.h>

struct Employee {
    char name[50];
    int id;
};

int main() {
    struct Employee emp = {"Alice", 1001};

    printf("Employee Name: %s\n", emp.name);
    printf("Employee ID: %d\n", emp.id);

    return 0;
}

Example 2: Nested Structures


#include <stdio.h>

struct Date {
    int day;
    int month;
    int year;
};

struct Event {
    char name[50];
    struct Date eventDate;
};

int main() {
    struct Event event = {"Conference", {15, 9, 2024}};

    printf("Event Name: %s\n", event.name);
    printf("Event Date: %d-%d-%d\n", event.eventDate.day, event.eventDate.month, event.eventDate.year);

    return 0;
}

Example 3: Array of Structures


#include <stdio.h>

struct Product {
    char name[50];
    float price;
};

int main() {
    struct Product products[2] = {
        {"Laptop", 999.99},
        {"Phone", 499.99}
    };

    for (int i = 0; i < 2; i++) {
        printf("Product Name: %s, Price: %.2f\n", products[i].name, products[i].price);
    }

    return 0;
}

Example 4: Union Basics


#include <stdio.h>

union Data {
    int intValue;
    float floatValue;
};

int main() {
    union Data data;

    data.intValue = 100;
    printf("Integer Value: %d\n", data.intValue);

    data.floatValue = 98.6;
    printf("Float Value: %.2f\n", data.floatValue);

    return 0;
}

Example 5: Union with Structs


#include <stdio.h>

union Info {
    int intValue;
    char charValue;
    float floatValue;
};

struct Student {
    char name[50];
    union Info info;
};

int main() {
    struct Student student = {"John", {100}};
    printf("Student Name: %s\n", student.name);
    printf("Union Integer Value: %d\n", student.info.intValue);

    student.info.floatValue = 3.14;
    printf("Union Float Value: %.2f\n", student.info.floatValue);

    return 0;
}
